// GetDiskAvailableSize 计算硬盘剩余可用大小，返回值单位是 MB
func GetDiskAvailableSize() int {
	stat := syscall.Statfs_t{}
	syscall.Statfs("/", &stat)

	all := stat.Blocks / 1024
	avail := stat.Bavail / 1024
	Log.Printf("[GetDiskAvailableSize] all = <%dmb>, avail = <%dmb>\n", all, avail)

	return int(avail)
}
